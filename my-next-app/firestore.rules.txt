rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow public read access to translations
    match /translations/{language} {
      allow read: if true;
      allow write: if request.auth != null 
        && exists(/databases/$(database)/documents/userRoles/$(request.auth.uid))
        && get(/databases/$(database)/documents/userRoles/$(request.auth.uid)).data.role in ['translator', 'admin'];
    }
    
    // User roles collection for permissions
    match /userRoles/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.isAdmin == true;
    }
  }
}
